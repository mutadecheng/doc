#把文件夹拷贝到image下
cd /root/image 


#构建镜像并推送到 Docker Hub 
cd git-client
docker build -t serset/git-client -t serset/git-client:1.0 .
 

#推送到镜像仓库
docker push serset/git-client
docker push serset/git-client:1.0
